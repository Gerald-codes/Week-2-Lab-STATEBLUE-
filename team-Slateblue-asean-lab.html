3.1 Read CSV

library(tidyverse)
library(readxl)
library(waldo)

3.2 Import the ASEAN Meat Production Data

meat_raw <- read_xlsx(
  path = "ASEAN-Statistical-Yearbook-2023.xlsx",
  sheet = "X.14",
  range = "A4:L36"
)

meat_raw

3.3 Tidy the Meat Production Data

meat_tidy <- meat_raw |>
  rename(animal = `Animal Type`) |>

  mutate(
    year = str_extract(as.character(animal), "^20[0-9]{2}$"),
    year = as.integer(year)
  ) |>
  fill(year) |>

  filter(
    animal %in%
      c(
        "Beef and Buffalo Meat",
        "Pig meat",
        "Mutton and Goat Meat",
        "Poultry meat"
      )
  ) |>

  select(-ASEAN) |>

  pivot_longer(
    cols = -c(year, animal),
    names_to = "country",
    values_to = "production"
  ) |>

  mutate(
    animal = recode(
      animal,
      "Beef and Buffalo Meat" = "beef and buffalo",
      "Pig meat" = "pig",
      "Mutton and Goat Meat" = "mutton and goat",
      "Poultry meat" = "poultry"
    ),
    production = as.numeric(production)
  ) |>

  select(year, animal, country, production) |>
  arrange(desc(year), animal, country)
head(meat_tidy, 6)
tail(meat_tidy, 6)
nrow(meat_tidy)

3.4 Correct Cambodia’s Mutton-and-Goat Entries

meat_tidy_original <- meat_tidy
meat_tidy <- meat_tidy_original |>
  mutate(
    production = if_else(
      country == "Cambodia" & animal == "mutton and goat",
      NA_real_,
      production
    )
  )
meat_tidy_verifier <- meat_tidy_original

meat_tidy_verifier[
  meat_tidy_verifier$country == "Cambodia" &
    meat_tidy_verifier$animal == "mutton and goat",
  "production"
] <- NA_real_

waldo::compare(meat_tidy, meat_tidy_verifier)
✔ No differences

3.5 Checkpoint: Compare meat_tidy to Reference Data

meat_ref <- read_csv("meat_tidy_reference.csv")

meat_tidy <- meat_tidy |>
  select(country, animal, year, production)

compare(meat_tidy, meat_ref, tolerance = 1e-6)
✔ No differences

3.6 Tidy the Population Data

pop_raw <- read_excel(
  "ASEAN-Statistical-Yearbook-2023.xlsx",
  sheet = "I.1",
  skip = 4,
  n_max = 10
)
pop_tidy <- pop_raw |>
  pivot_longer(
    cols = -Country,
    names_to = "year",
    values_to = "population"
  ) |>
  mutate(
    year = as.integer(year),
    population = as.numeric(population)

merged <- meat_tidy |>
  left_join(pop_tidy, by = c("country", "year")) |>
  rename(pop = population) |>

3.11 Visualize the Result

ggplot(
  merged,
  aes(x = year, y = meat_per_cap, color = animal)
) +
  geom_line() +
  geom_point() +
  labs(
    x = NULL,
    y = "Production (kg per person)",
    color = "Animal",
    title = "Per-Capita Meat Production in ASEAN, 2018–2021",
    caption = "Source: ASEAN Statistical Yearbook 2023"
  ) +
  xlim(2017.9, 2021.1) +
  scale_y_log10(
    breaks = c(0.01, 0.1, 1, 10, 100),
    labels = c("0.01", "0.1", "1", "10", "100")
  ) +
  facet_wrap(~country) +
  theme(
    legend.position = "inside",
    legend.position.inside = c(1, 0),
    legend.justification.inside = c(1, 0.05)
  )
